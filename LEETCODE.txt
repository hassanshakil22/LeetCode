#include <climits>

class Solution {
public:
    int reverse(int x) {
        long reversed = 0;
        while (x != 0) {
            reversed = reversed * 10 + (x % 10);
            
            // Check for overflow before casting to int
            if (reversed > INT_MAX || reversed < INT_MIN) {
                return 0;  
            }
            
            x /= 10;
        }
        return (int)reversed;
    }
};

class Solution {
public:
    double findMedianSortedArrays(std::vector<int>& nums1, std::vector<int>& nums2) {
        int n1 = nums1.size();
        int n2 = nums2.size();
        int n = n1 + n2;
        vector<int> nums(n);
        

        for (int i = 0; i < n1; i++) {
            nums[i] = nums1[i];
        }
        for (int i = 0; i < n2; i++) {
            nums[i + n1] = nums2[i];
        }

        sort(nums.begin(), nums.end());
        if (n % 2 == 0) { 
            return (nums[n / 2 - 1] + nums[n / 2]) / 2.0;
        } else {  
            return nums[n / 2];
        }
    }
};

    int n = matrix.size();
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            swap(matrix[i][j], matrix[j][i]); 
        }
    }
    for (int i = 0; i < n; ++i) {
        reverse(matrix[i].begin(), matrix[i].end()); // Reverse the row
    }

class Solution {
public:
    int maxArea(vector<int>& height) {
        int left = 0;
        int right = height.size() - 1;
        int max_area = 0;
        while (left < right) {
            int current_area = min(height[left], height[right]) * (right - left);
            max_area = max(max_area, current_area);
            if (height[left] < height[right]) {
                ++left;
            } else {
                --right;
            }
        }
        return max_area;
    }
};

    int i = m - 1;
    int j = n - 1;
    int k = m + n - 1;
    while (i >= 0 && j >= 0) {
        if (nums1[i] > nums2[j]) {
            nums1[k] = nums1[i];
            i--;
        } else {
            nums1[k] = nums2[j];
            j--;
        }
        k--;
    }
    while (j >= 0) {
        nums1[k] = nums2[j];
        j--;
        k--;
    }
    }
};

class Solution {
public:
    int computeArea(int A, int B, int C, int D, int E, int F, int G, int H) {
        
        int area1 = (C - A) * (D - B);
        int area2 = (G - E) * (H - F);
        
        int overlap_width = max(0, min(C, G) - max(A, E));
        int overlap_height = max(0, min(D, H) - max(B, F));
        
        int overlap_area = overlap_width * overlap_height;
        
        return area1 + area2 - overlap_area;
    }
};

class Solution {

        int totalSum = 0;
        int n = arr.size();
        
        // Generate all odd-length subarrays
        for (int start = 0; start < n; ++start) {
            for (int length = 1; start + length <= n; length += 2) {
                // Sum the elements of the subarray
                for (int i = start; i < start + length; ++i) {
                    totalSum += arr[i];
                }
            }
        }
        
        return totalSum;
    }
};

#include <vector>
#include <algorithm>

class Solution {
public:
    int maxContainers(std::vector<int>& weights, int limit) {
        std::sort(weights.begin(), weights.end());  // Sort weights in ascending order
        int count = 0, totalWeight = 0;
        
        for (int weight : weights) {
            if (totalWeight + weight <= limit) {
                totalWeight += weight;
                count++;
            } else {
                break;  // Stop if we exceed the weight limit
            }
        }
        
        return count;
    }
};
